def moyenne_coef(dm, ds):
    coeff_dm = 1
    coeff_ds = 3
    somme_coeff = coeff_dm + coeff_ds
    moyenne = (coeff_dm*dm + coeff_ds*ds) / somme_coeff
    return moyenne

ma_moyenne = moyenne_coef(12 , 10)
print(ma_moyenne)
